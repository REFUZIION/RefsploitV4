RemoveDoorsAndLasers = function()
	workspace.Banks.Bank.Door:Destroy()
	workspace.Banks.Bank.TriggerDoor:Destroy()
	workspace.Banks.Bank.Lasers:Destroy()
	workspace.Doors:Destroy()
end
FindOpenHeli = function()
	for i,v in pairs(workspace.Vehicles:children()) do
		if v.Name == "Heli" then
			if v.Seat.Player.Value == false then
				Plr.Character.HumanoidRootPart.CFrame = v.Seat.CFrame + Vector3.new(0,3,0)
			end
		end
	end
end
FindOpenCar = function()
	for i,v in pairs(workspace.Vehicles:children()) do
		if v.Name == "Camaro" then
			if v.Seat.Player.Value == false then
				Plr.Character.HumanoidRootPart.CFrame = v.Seat.CFrame + Vector3.new(0,4,0)
			end
		end
	end
end
